Medication Tracker

The Medication Tracker allows users to input their medication details, track their intake, and receive reminders.

Code Implementation:


document.getElementById('medication-form').addEventListener('submit', function(event) {
    event.preventDefault(); // Prevent form submission

    // Get medication details
    const medicationName = document.getElementById('medication-name').value;
    const dosage = document.getElementById('dosage').value;
    const frequency = document.getElementById('frequency').value;
    const startDate = document.getElementById('start-date').value;
    const endDate = document.getElementById('end-date').value;

    // Create medication object
    const medication = {
        name: medicationName,
        dosage: dosage,
        frequency: frequency,
        startDate: startDate,
        endDate: endDate
    };

    // Store medication in local storage
    let medications = JSON.parse(localStorage.getItem('medications')) || [];
    medications.push(medication);
    localStorage.setItem('medications', JSON.stringify(medications));

    // Clear form fields
    document.getElementById('medication-form').reset();

    // Display updated medication list
    displayMedications();
});

Diet Recommendations

Users can select their health conditions to receive personalized diet plans.

Code Implementation:


document.getElementById('generate-diet').addEventListener('click', function() {
    const condition = document.getElementById('health-condition').value;
    if (!condition) return alert('Please select a health condition');
    
    const dietPlans = {
        diabetes: {
            breakfast: "Oatmeal with nuts and berries",
            lunch: "Grilled chicken with quinoa and vegetables",
            dinner: "Salmon with steamed broccoli"
        },
        hypertension: {
            breakfast: "Greek yogurt with banana",
            lunch: "Lentil soup with whole grain bread",
            dinner: "Grilled fish with sweet potato"
        }
    };
    
    const plan = dietPlans[condition] || {
        breakfast: "Balanced meal with protein and fiber",
        lunch: "Lean protein with complex carbs",
        dinner: "Light meal with vegetables"
    };
    
    const dietPlan = document.getElementById('diet-plan');
    dietPlan.innerHTML = `
        

            
Breakfast

            
${plan.breakfast}

        

        

            
Lunch

            
${plan.lunch}

        

        

            
Dinner

            
${plan.dinner}

        

    `;
});
